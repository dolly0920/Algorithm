수 124를 뒤집으면 421이 되고 이 두 수를 합하면 545가 된다. 124와 같이 원래 수와 뒤집은 수를 합한 수가 좌우 대칭이 되는지 테스트하는 프로그램을 작성하시오.
----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
// My solution
#include <stdio.h>
 
int N, M, sym;

int reverse_number(int n)	// 뒤집은 숫자 n 리턴
{
	int num = 0;

	while(n!=0)
	{
		num *= 10;
		num += (n%10);
		n /= 10;
	}
	
	return num;
}
 
int main(void)
{
      int i,tmp;
 
      // 입력받는 부분
      scanf("%d", &N);
 
      for(i=0; i<N; i++)
      {
            // 입력받는 부분
            scanf("%d", &M);
 
            // 여기서부터 작성
			tmp = reverse_number(M) + M;
			if(tmp == reverse_number(tmp))	sym = 1;
			else	sym = 0;
 
            // 출력하는 부분
            if(sym == 1)      printf("YES\n");
            else  printf("NO\n");
      }
 
      return 0;
}
